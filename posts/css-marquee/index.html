input file

site2.css:

@theme {
  --animate-marquee1: marquee1 15s linear infinite;
  --animate-marquee2: marquee2 15s linear infinite;
  
  @keyframes marquee1 {
        0% {
          transform: translateX(0%);
        }
        100% {
          transform: translateX(-100%);
        }
    }
    @keyframes marquee2 {
        0% {
          transform: translateX(100%);
        }
        100% {
          transform: translateX(0%);
        }
    }

}
